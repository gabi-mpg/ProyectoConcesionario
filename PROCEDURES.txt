drop procedure if exists sp_eliminarcliente;
*
CREATE PROCEDURE sp_eliminarcliente(IN idcliente varchar(9))
BEGIN
UPDATE t_clientes SET existe = 0 where DNI like idcliente;
END
*
drop procedure if exists sp_eliminarmoto;
*
CREATE PROCEDURE sp_eliminarmoto(IN idmoto varchar(15))
BEGIN
UPDATE t_motos SET existe = 0 where Matricula like idmoto;
END
*

*
CREATE PROCEDURE sp_eliminarusuario(IN iduser varchar(30))

UPDATE t_usuarios SET existe = 0 where nick like iduser;
END
